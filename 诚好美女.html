from flask import Flask, render_template_string app = Flask(__name__) poem_html = """ 致诚好

致诚好

在时光长河悠然流淌，
有个名字如星光闪亮，
诚好，恰似春日的暖阳，
绽放在我心的每一寸地方。

她的眼眸，似深邃海洋，
藏着星辰，泛着温柔波光，
睫毛轻颤，像蝶翼舒张，
扇动的微风，吹进我心房。

那弯弯的眉，似月牙清朗，
浅笑时，嘴角梨涡荡漾，
仿佛是命运精心的雕章，
刻画出这绝美的模样。

她的秀发，如乌瀑飞扬，
散发着梦幻的迷人芬芳，
每一步轻盈，似音符奏响，
在岁月里谱出优雅乐章。

诚好啊，你是美的诗行，
在尘世闪耀独特的光芒，
愿时光温柔把你守望，
让你的美，永恒地绽放。
""" @app.route('/') def display_poem(): return render_template_string(poem_html) if __name__ == '__main__': app.run(debug=True)
