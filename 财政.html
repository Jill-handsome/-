class FinanceManager: def __init__(self): self.transactions = [] self.balance = 0.0 def record_income(self, amount, description="Income"): self.transactions.append({"type": "income", "amount": amount, "description": description}) self.balance += amount print(f"Income recorded: {amount} ({description})") def record_expense(self, amount, description="Expense"): self.transactions.append({"type": "expense", "amount": amount, "description": description}) self.balance -= amount print(f"Expense recorded: {amount} ({description})") def get_balance(self): return self.balance def generate_report(self): print("\nFinance Report:") print(f"Current Balance: {self.balance}") print("Transactions:") for transaction in self.transactions: print(f" {transaction['type'].capitalize()}: {transaction['amount']} ({transaction['description']})") def main(): fm = FinanceManager() while True: print("\nFinance Manager") print("1. Record Income") print("2. Record Expense") print("3. View Balance") print("4. Generate Report") print("5. Exit") choice = input("Please enter your choice: ") if choice == '1': amount = float(input("Enter the income amount: ")) description = input("Enter a description (optional): ") or "Income" fm.record_income(amount, description) elif choice == '2': amount = float(input("Enter the expense amount: ")) description = input("Enter a description (optional): ") or "Expense" fm.record_expense(amount, description) elif choice == '3': print(f"\nCurrent Balance: {fm.get_balance()}") elif choice == '4': fm.generate_report() elif choice == '5': print("Exiting Finance Manager. Goodbye!") break else: print("Invalid choice. Please try again.") if __name__ == "__main__": main()
